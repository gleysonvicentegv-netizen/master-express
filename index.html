let autocompleteColeta, autocompleteEntrega; function initAutocomplete() { const coletaInput = document.getElementById('coleta'); const entregaInput = document.getElementById('entrega'); autocompleteColeta = new google.maps.places.Autocomplete(coletaInput, { types: ['geocode'] }); autocompleteEntrega = new google.maps.places.Autocomplete(entregaInput, { types: ['geocode'] }); // Só calcula quando o usuário selecionar uma sugestão válida autocompleteColeta.addListener('place_changed', calcularValor); autocompleteEntrega.addListener('place_changed', calcularValor); // Permitir digitar livremente sem travar coletaInput.addEventListener('input', () => { document.getElementById('valorEntrega').innerText = 'Valor da entrega: R$0,00'; }); entregaInput.addEventListener('input', () => { document.getElementById('valorEntrega').innerText = 'Valor da entrega: R$0,00'; }); } function calcularValor() { const coletaPlace = autocompleteColeta.getPlace(); const entregaPlace = autocompleteEntrega.getPlace(); // Se algum endereço não estiver selecionado corretamente, não calcula if (!coletaPlace || !entregaPlace || !coletaPlace.geometry || !entregaPlace.geometry) return; const service = new google.maps.DistanceMatrixService(); service.getDistanceMatrix({ origins: [coletaPlace.formatted_address], destinations: [entregaPlace.formatted_address], travelMode: 'DRIVING', }, function(response, status){ if(status === 'OK'){ let distanciaMetros = response.rows[0].elements[0].distance.value; let distanciaKm = distanciaMetros / 1000; let valor = 10; if(distanciaKm > 3){ valor += (distanciaKm - 3) * 1.8; } valor = valor.toFixed(2); document.getElementById('valorEntrega').innerText = `Valor da entrega: R$${valor}`; } }); } initAutocomplete();
